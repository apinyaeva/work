function euclidean_gcd(a::Int, b::Int)
a, b = abs(a), abs(b)
while b != 0
a, b = b, a % b
end
return a
end

function binary_gcd(a::Int, b::Int)
a, b = abs(a), abs(b)
a == 0 && return b
b == 0 && return a
shift = 0
while iseven(a) && iseven(b)
a = 2
b = 2
shift += 1
end
while a != 0
while iseven(a)
a = 2
end
while iseven(b)
b = 2
end
if a >= b
a = a - b
else
b = b - a
end
end
return b << shift # b * 2^shift
end

function extended_euclidean(a::Int, b::Int)
a, b = abs(a), abs(b)
if b == 0
return (a, 1, 0)
end
x0, x1 = 1, 0
y0, y1 = 0, 1
while b != 0
q = a b
a, b = b, a % b
x0, x1 = x1, x0 - q * x1
y0, y1 = y1, y0 - q * y1
end
return (a, x0, y0)
end

function extended_binary_gcd(a::Int, b::Int)
a, b = abs(a), abs(b)
if a == 0
return (b, 0, 1)
elseif b == 0
return (a, 1, 0)
end
g = 1
while iseven(a) && iseven(b)
a = 2
b = 2
g *= 2
end
u, v = a, b
A, B, C, D = 1, 0, 0, 1
while u != 0
while iseven(u)
u = 2
if iseven(A) && iseven(B)
A = 2
B = 2
else
A = (A + b) 2
B = (B - a) 2
end
end
while iseven(v)
v = 2
if iseven(C) && iseven(D)
C = 2
D = 2
else
C = (C + b) 2
D = (D - a) 2
end
end
if u >= v
u -= v
A -= C
B -= D
else
v -= u
C -= A
D -= B
end
end
d = g * v
return (d, C, D)
end
